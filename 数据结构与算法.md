## 最小的K个数
1. 快排法(允许改变输入数据)
```cpp
class Solution {
public:
    vector<int> getLeastNumbers(vector<int>& arr, int k) {
        vector<int> res;
        if(arr.empty() || k < 1 || k > arr.size()) return res;
        int index = 0, left = 0, right = arr.size()-1;
        index = partition(arr, left, right);
        while(index != k-1)
        {
            if(k-1 < index )
            {
                right = index - 1;
                index = partition(arr, left, right);
            }
            else if(k-1 > index)
            {
                left = index + 1;
                index = partition(arr, left, right);
            }
        }
        for(int i=0; i<k; ++i)
        {
            res.push_back(arr[i]);
        }
        return res;
  
    }
    int partition(vector<int>& arr, int left, int right)
    {
        int temp = arr[left];
        while(left < right)
        {
            while(left < right && arr[right] > temp) --right;
            arr[left] = arr[right];
            while(left < right && arr[left] <= temp) ++left;
            arr[right] = arr[left];
        }
        arr[left] = temp;
        return left;
    }
};
```
2. 容器法(不改变输入数据)
```cpp
class Solution {
public:
    vector<int> getLeastNumbers(vector<int>& arr, int k) {
        vector<int> res;
        if(arr.empty() || k < 1 || k > arr.size()) return res;
        multiset<int> least;

        for(auto data = arr.begin(); data != arr.end(); ++data)
        {
            if(least.size() < k)
            {
                least.insert(*data);
            }
            else
            {
                auto max = least.end();
                --max;
                if(*data < *max)
                {
                    least.erase(max);
                    least.insert(*data);
                }  
            }
        }
        for(auto &it : least)
        {
            res.push_back(it);
        }
        return res;
    }
};
```


## 判断平衡二叉树
二叉树递归 求解套路
```cpp
class Solution {
public:
    class RetType
    {
    public:
        RetType(bool isbal, int height)
        {
            this->isbal = isbal;
            this->height = height;
        }
        bool isbal;
        int height;
    };

    bool isBalanced(TreeNode* root) {
        return depth(root).isbal;
    }

    RetType depth(TreeNode* root) 
    {
        if (root == nullptr) {
            return RetType(true, 0);
        }       
        RetType left = depth(root->left);
        if(left.isbal == false) {
            return RetType(false, 0);
        }            
        RetType right = depth(root->right);
        if(right.isbal  == false) {
            return RetType(false, 0);
        }
        if(abs(left.height - right.height) > 1) { 
            return RetType(false, 0);
        }
        return RetType(true, max(left.height, right.height) + 1);
    }
};
```
