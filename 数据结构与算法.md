## Partition
    int partition(vector<int>& arr, int left, int right)  
    {  
        int temp = arr[left];  
        while(left < right)  
        {  
            while(left < right && arr[right] > temp) --right;  
            arr[left] = arr[right];  
            while(left < right && arr[left] <= temp) ++left;  
            arr[right] = arr[left];  
        }  
        arr[left] = temp;  
        return left;  
    }
