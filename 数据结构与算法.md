### 冒泡排序
```cpp
void BubbleSort(vector<int> &nums) {
    for (int end = nums.size() - 1; end > 0; --end) {
        for (int i = 0; i < end; ++i) {
            if (nums[i] > nums[i + 1]) {
                swap(nums[i], nums[i + 1]);
            }
        }
    }
}
```
### 选择排序
```cpp
oid SelectSort(vector<int> &nums) {
    for (int start = 0; start < nums.size(); ++start) {
        int minindex = start;
        for (int i = start; i < nums.size(); ++i) {
            minindex = nums[i] < nums[minindex] ? i : minindex;
        }
        swap(nums[start], nums[minindex]);
    }
}
```

```cpp
void InsertSort(vector<int> &nums){
    for(int i = 1; i < nums.size(); ++i){
        for(int j = i - 1; j > 0 && nums[j] > nums[j+1]; --j){
            swap(nums[j],nums[j+1]);
        }
    }
}
```

```cpp
vector<int> partion(vector<int> &nums, int left, int right) {
    int less = left - 1;
    int more = right + 1;
    int cur = left;
    int temp = nums[right];
    while (cur < more) {
        if (nums[cur] < temp) {
            swap(nums[++less], nums[cur++]);
        } else if (nums[cur] > temp) {
            swap(nums[--more], nums[cur]);
        } else if (nums[cur] == temp) {
            ++cur;
        }
    }
    return {less + 1, more - 1};
}

void QuickSort(vector<int> &nums, int left, int right) {
    if(left < right){
        auto p = partion(nums, left, right);
        QuickSort(nums, left, p[0] - 1);
        QuickSort(nums, p[1] + 1, right);
    }
}
```

## 二分查找
```cpp
int binary_search(int[] nums, int target) {
    int left = 0, right = nums.length - 1; 
    while(left <= right) {
        int mid = left + (right - left) / 2;
        if (nums[mid] < target) {
            left = mid + 1;
        } else if (nums[mid] > target) {
            right = mid - 1; 
        } else if(nums[mid] == target) {
            // 直接返回
            return mid;
        }
    }
    // 直接返回
    return -1;
}

int left_bound(int[] nums, int target) {
    int left = 0, right = nums.length - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (nums[mid] < target) {
            left = mid + 1;
        } else if (nums[mid] > target) {
            right = mid - 1;
        } else if (nums[mid] == target) {
            // 别返回，锁定左侧边界
            right = mid - 1;
        }
    }
    // 最后要检查 left 越界的情况
    if (left >= nums.length || nums[left] != target)
        return -1;
    return left;
}


int right_bound(int[] nums, int target) {
    int left = 0, right = nums.length - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (nums[mid] < target) {
            left = mid + 1;
        } else if (nums[mid] > target) {
            right = mid - 1;
        } else if (nums[mid] == target) {
            // 别返回，锁定右侧边界
            left = mid + 1;
        }
    }
    // 最后要检查 right 越界的情况
    if (right < 0 || nums[right] != target)
        return -1;
    return right;
}
```

## 最小的K个数
1. 快排法(允许改变输入数据)
```cpp
class Solution {
public:
    vector<int> getLeastNumbers(vector<int>& arr, int k) {
        vector<int> res;
        if(arr.empty() || k < 1 || k > arr.size()) return res;
        int index = 0, left = 0, right = arr.size()-1;
        index = partition(arr, left, right);
        while(index != k-1)
        {
            if(k-1 < index )
            {
                right = index - 1;
                index = partition(arr, left, right);
            }
            else if(k-1 > index)
            {
                left = index + 1;
                index = partition(arr, left, right);
            }
        }
        for(int i=0; i<k; ++i)
        {
            res.push_back(arr[i]);
        }
        return res;
  
    }
    int partition(vector<int>& arr, int left, int right)
    {
        int temp = arr[left];
        while(left < right)
        {
            while(left < right && arr[right] > temp) --right;
            arr[left] = arr[right];
            while(left < right && arr[left] <= temp) ++left;
            arr[right] = arr[left];
        }
        arr[left] = temp;
        return left;
    }
};
```

## 判断平衡二叉树
二叉树递归 求解套路
```cpp
class Solution {
public:
    class RetType
    {
    public:
        RetType(bool isbal, int height)
        {
            this->isbal = isbal;
            this->height = height;
        }
        bool isbal;
        int height;
    };

    bool isBalanced(TreeNode* root) {
        return depth(root).isbal;
    }

    RetType depth(TreeNode* root) 
    {
        if (root == nullptr) {
            return RetType(true, 0);
        }       
        RetType left = depth(root->left);
        if(left.isbal == false) {
            return RetType(false, 0);
        }            
        RetType right = depth(root->right);
        if(right.isbal  == false) {
            return RetType(false, 0);
        }
        if(abs(left.height - right.height) > 1) { 
            return RetType(false, 0);
        }
        return RetType(true, max(left.height, right.height) + 1);
    }
};
```
